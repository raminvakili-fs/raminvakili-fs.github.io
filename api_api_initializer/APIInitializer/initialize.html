Injector initialize({UniqueKey? uniqueKey, bool isMock = false}) {
  final UniqueKey apiUniqueKey = uniqueKey ?? UniqueKey();

  return Injector.getInjector()
    ..map<BaseAPI>(
      factoryFunction: (_) =>
          isMock ? MockAPI(apiUniqueKey) : BinaryAPI(apiUniqueKey),
      isSingleton: true,
    );
}
